export const imageModules = import.meta.glob('./DI.PREDICTION/*.{png,jpeg,jpg}', { eager: true });

export const images = Object.entries(imageModules)
  .map(([path, mod]) => {
    const filename = path.split('/').pop();
    const match = filename.match(/^(\d+)\.(.+)\.\w+$/);
    if (!match) return null;
    const num = parseInt(match[1]);
    const title = match[2].replace(/_/g, ' ');
    const src = typeof mod.default === 'string' ? mod.default : mod.default?.src;
    const category = categoryMap[num] || 'image';
    const description = descriptionMap[num] || '';
    return { src, title: `${num}. ${title}`, num, category, description };
  })
  .filter(Boolean)
  .sort((a, b) => a.num - b.num);
